def self.draw(map)
  map.jammit "/#{Jammit.package_path}/:package.:extension", {
    :controller => 'jammit',
    :action => 'package',
    :requirements => {
      # A hack to allow extension to include "."
      :extension => /.+/
    }
  }
end
